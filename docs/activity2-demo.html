‚úÖ Secure Authentication Implementation

‚úÖ Secure Implementation: This code implements proper authentication security
public class SecureAuthentication { private readonly IPasswordHasher passwordHasher; private readonly ISessionManager sessionManager; private readonly IRateLimiter rateLimiter; private readonly ILogger logger; public async Task<AuthResult> Login_Secure(string username, string password, string ipAddress, string userAgent) { // ‚úÖ Rate limiting prevents brute force if (!await rateLimiter.IsAllowed(ipAddress, "login")) return new AuthResult { Success = false, Message = "Rate limit exceeded" }; // ‚úÖ Input validation if (!ValidateInput(username, password)) return new AuthResult { Success = false, Message = "Invalid input" }; // ‚úÖ Secure user lookup var user = await userRepository.GetUserByUsername(username); if (user == null || !user.IsActive) { // ‚úÖ Constant-time response prevents user enumeration await Task.Delay(100); logger.LogWarning($"Login attempt for non-existent user: {username}"); return new AuthResult { Success = false, Message = "Invalid credentials" }; } // ‚úÖ Secure password verification with salt if (!passwordHasher.VerifyPassword(password, user.PasswordHash, user.Salt)) { await rateLimiter.RecordFailedAttempt(ipAddress, username); logger.LogWarning($"Failed login attempt: {username} from {ipAddress}"); return new AuthResult { Success = false, Message = "Invalid credentials" }; } // ‚úÖ Check for compromised passwords if (await IsPasswordCompromised(password)) { logger.LogWarning($"Compromised password used by: {username}"); return new AuthResult { Success = false, Message = "Password appears in data breaches. Please change it.", RequiresPasswordChange = true }; } // ‚úÖ Secure session creation var session = await sessionManager.CreateSecureSession(user.Id, ipAddress, userAgent); // ‚úÖ Update last login user.LastLogin = DateTime.UtcNow; await userRepository.UpdateUser(user); logger.LogInfo($"Successful login: {username}"); return new AuthResult { Success = true, SessionId = session.Id, RequiresMFA = user.MFAEnabled && !session.MFAVerified, User = user }; } // ‚úÖ SECURE: Multi-factor authentication public async Task<MFAResult> VerifyMFA_Secure(string sessionId, string mfaCode) { var session = await sessionManager.GetSession(sessionId); if (session?.IsValid != true) return new MFAResult { Success = false, Message = "Invalid session" }; var user = await userRepository.GetUser(session.UserId); if (user?.MFAEnabled != true) return new MFAResult { Success = false, Message = "MFA not enabled" }; // ‚úÖ TOTP verification with time window tolerance bool isValidTOTP = totpValidator.ValidateCode(user.MFASecret, mfaCode, timeWindowsToCheck: 2); if (isValidTOTP) { session.MFAVerified = true; session.MFAVerifiedAt = DateTime.UtcNow; await sessionManager.UpdateSession(session); logger.LogInfo($"MFA verified for user: {user.Username}"); return new MFAResult { Success = true }; } logger.LogWarning($"Invalid MFA attempt for user: {user.Username}"); return new MFAResult { Success = false, Message = "Invalid MFA code" }; } // ‚úÖ SECURE: Role-based authorization public async Task<bool> DeleteUser_Secure(string sessionId, string targetUserId) { // ‚úÖ Session validation var session = await sessionManager.GetSession(sessionId); if (!sessionManager.IsValidSession(session)) return false; var currentUser = await userRepository.GetUser(session.UserId); // ‚úÖ Authorization check if (!authorizationService.HasPermission(currentUser, "DELETE_USER")) { logger.LogWarning($"Unauthorized delete attempt by: {currentUser.Username}"); return false; } // ‚úÖ Additional validation for admin operations if (currentUser.Role != "SuperAdmin" && targetUserId == currentUser.Id) { logger.LogWarning($"User cannot delete themselves: {currentUser.Username}"); return false; } var targetUser = await userRepository.GetUser(targetUserId); if (targetUser == null) return false; // ‚úÖ Audit logging logger.LogInfo($"User deleted: {targetUser.Username} by {currentUser.Username}"); await userRepository.DeleteUser(targetUserId); return true; } }

// ‚úÖ PBKDF2 with SHA-256 and high iteration count public static string HashPassword(string password, out byte[] salt) { salt = RandomNumberGenerator.GetBytes(32); using var pbkdf2 = new Rfc2898DeriveBytes(password, salt, 100000, HashAlgorithmName.SHA256); return Convert.ToBase64String(pbkdf2.GetBytes(32)); }

2. Secure Session Management

// ‚úÖ Cryptographically secure session tokens public string GenerateSessionToken() { var tokenBytes = RandomNumberGenerator.GetBytes(32); return Convert.ToBase64String(tokenBytes); } // ‚úÖ Session validation with timeout public bool IsValidSession(Session session) { return session != null && session.ExpiresAt > DateTime.UtcNow && !session.IsRevoked && session.MFAVerified; }

3. Multi-Factor Authentication (TOTP)

// ‚úÖ Time-based One-Time Password implementation public bool ValidateTOTP(string secret, string userCode) { var totp = new Totp(Base32Encoding.ToBytes(secret)); var validCodes = new[] { totp.ComputeTotp(DateTime.UtcNow.AddSeconds(-30)), totp.ComputeTotp(DateTime.UtcNow), totp.ComputeTotp(DateTime.UtcNow.AddSeconds(30)) }; return validCodes.Contains(userCode); }
